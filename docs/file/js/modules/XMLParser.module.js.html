    // @flow
import XMLNode from "./XMLNode.module";

/**
 * @access private
 * @typedef {Tuple<EOF, ?XMLNode>} State
 */
type State = [EOF, ?XMLNode];

/**
 * @access private
 * @typedef {boolean} EOF
 */
type EOF = boolean;

/**
 * @access public
 * @desc the class in charge of parsing XML string
 */
export default class XMLParser {
	_xml_str: string;

	_progress: number;
	
	/**
	 * @access public
	 * @param {string} xml_str - XML string 
	 */
	constructor(xml_str: string) {
		this._xml_str = xml_str;
		this._progress = -1;
	}

	/**
	 * @access public
	 * @desc parse the XML string
	 * @return {XMLNode[]} - XML AST
	 */
	parse(): XMLNode[] {
		let opened_tags: XMLNode[] = [];
		return [];
	}

	/**
	 * @access private
	 */
	_parseTag(): State {
		this._skipWhiteSpaces();
		const char: string = this._xml_str.charAt(this._progress);
		return [false, new XMLNode("div")];
	}

	/**
	 * @access private
	 */
	_skipWhiteSpaces(): State {
		const whiteSpaces: string[] = [" ", "\n", "\t", "\r\n"];
		while (true) {
			let c: string = this._xml_str.charAt(++this._progress);
			this._progress ++;
			if (c === "") return [true, null];
			if (whiteSpaces.indexOf(c) === -1) break;
		}
		this._progress --;
		return [false, null];
	}
}
